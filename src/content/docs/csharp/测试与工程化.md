## 1. 单元测试

### 1.1 xUnit

```csharp
// 安装: dotnet add package xunit
public class CalculatorTests
{
    private readonly Calculator _calc = new();

    [Fact]
    public void Add_TwoNumbers_ReturnsSum()
    {
        // Arrange
        int a = 3, b = 5;

        // Act
        int result = _calc.Add(a, b);

        // Assert
        Assert.Equal(8, result);
    }

    // 参数化测试
    [Theory]
    [InlineData(1, 2, 3)]
    [InlineData(-1, 1, 0)]
    [InlineData(0, 0, 0)]
    [InlineData(100, 200, 300)]
    public void Add_VariousInputs_ReturnsExpected(int a, int b, int expected)
    {
        Assert.Equal(expected, _calc.Add(a, b));
    }

    // 使用 MemberData
    public static TheoryData<int, int, int> AddData => new()
    {
        { 1, 2, 3 },
        { -1, 1, 0 },
        { 100, -50, 50 }
    };

    [Theory]
    [MemberData(nameof(AddData))]
    public void Add_MemberData_ReturnsExpected(int a, int b, int expected)
    {
        Assert.Equal(expected, _calc.Add(a, b));
    }

    // 异常测试
    [Fact]
    public void Divide_ByZero_ThrowsDivideByZeroException()
    {
        Assert.Throws<DivideByZeroException>(() => _calc.Divide(10, 0));
    }

    // 异步测试
    [Fact]
    public async Task FetchDataAsync_ReturnsExpected()
    {
        var result = await _service.FetchDataAsync();
        Assert.NotNull(result);
        Assert.NotEmpty(result);
    }
}
```

### 1.3 Moq - 模拟框架

```csharp
// 安装: dotnet add package Moq
public class OrderServiceTests
{
    private readonly Mock<IOrderRepository> _repoMock;
    private readonly Mock<IEmailService> _emailMock;
    private readonly OrderService _service;

    public OrderServiceTests()
    {
        _repoMock = new Mock<IOrderRepository>();
        _emailMock = new Mock<IEmailService>();
        _service = new OrderService(_repoMock.Object, _emailMock.Object);
    }

    [Fact]
    public async Task CreateOrder_ValidOrder_SavesAndNotifies()
    {
        // Arrange
        var order = new Order { Id = 1, CustomerEmail = "test@test.com" };
        _repoMock.Setup(r => r.SaveAsync(It.IsAny<Order>()))
                 .ReturnsAsync(order);

        // Act
        var result = await _service.CreateOrderAsync(order);

        // Assert
        Assert.NotNull(result);
        _repoMock.Verify(r => r.SaveAsync(It.IsAny<Order>()), Times.Once);
        _emailMock.Verify(
            e => e.SendAsync(order.CustomerEmail, It.IsAny<string>()),
            Times.Once);
    }

    // 验证属性
    [Fact]
    public void GetConfig_ReturnsConfiguredValue()
    {
        var mockConfig = new Mock<IConfiguration>();
        mockConfig.Setup(c => c["ApiKey"]).Returns("test-key");

        var value = mockConfig.Object["ApiKey"];
        Assert.Equal("test-key", value);
    }

    // 抛出异常
    [Fact]
    public async Task CreateOrder_RepoThrows_PropagatesException()
    {
        _repoMock.Setup(r => r.SaveAsync(It.IsAny<Order>()))
                 .ThrowsAsync(new DbUpdateException());

        await Assert.ThrowsAsync<DbUpdateException>(
            () => _service.CreateOrderAsync(new Order()));
    }

    // 回调
    [Fact]
    public async Task CreateOrder_AssignsId()
    {
        Order? saved = null;
        _repoMock.Setup(r => r.SaveAsync(It.IsAny<Order>()))
                 .Callback<Order>(o => { o.Id = 42; saved = o; })
                 .ReturnsAsync((Order o) => o);

        var result = await _service.CreateOrderAsync(new Order());
        Assert.Equal(42, result.Id);
    }
}
```

```csharp
public class DatabaseIntegrationTests : IAsyncLifetime
{
```

```csharp
// 安装: dotnet add package BenchmarkDotNet
[MemoryDiagnoser]
[RankColumn]
public class StringConcatBenchmarks
{
    private const int Iterations = 1000;
    private readonly string[] _parts = Enumerable.Range(0, Iterations)
        .Select(i => $"part{i}").ToArray();

    [Benchmark(Baseline = true)]
    public string StringConcat()
    {
        var result = "";
        foreach (var part in _parts)
            result += part;
        return result;
    }

    [Benchmark]
    public string StringBuilder()
    {
        var sb = new StringBuilder();
        foreach (var part in _parts)
            sb.Append(part);
        return sb.ToString();
    }

    [Benchmark]
    public string StringJoin() => string.Join("", _parts);

    [Benchmark]
    public string StringConcat2() => string.Concat(_parts);
}

// 运行
BenchmarkRunner.Run<StringConcatBenchmarks>();

// dotnet run -c Release  // 必须使用 Release 模式
```

## 4. Source Generator

### 4.1 概述

Source Generator 在编译期生成代码，替代运行时反射：

```csharp
// 项目文件 (.csproj)
// <Project Sdk="Microsoft.NET.Sdk">
//   <PropertyGroup>
//     <TargetFramework>netstandard2.0</TargetFramework>
//     <EnforceExtendedAnalyzerRules>true</EnforceExtendedAnalyzerRules>
//   </PropertyGroup>
//   <ItemGroup>
//     <PackageReference Include="Microsoft.CodeAnalysis.Analyzers" Version="*" />
//     <PackageReference Include="Microsoft.CodeAnalysis.CSharp" Version="*" />
//   </ItemGroup>
// </Project>

[Generator]
public class AutoNotifyGenerator : IIncrementalGenerator
{
    public void Initialize(IncrementalGeneratorInitializationContext context)
    {
        var provider = context.SyntaxProvider
            .CreateSyntaxProvider(
                predicate: (node, _) => node is FieldDeclarationSyntax,
                transform: (ctx, _) => GetFieldInfo(ctx))
            .Where(f => f is not null);

        context.RegisterSourceOutput(provider, (spc, field) =>
        {
            if (field is null) return;
            var source = GenerateProperty(field.Value);
            spc.AddSource($"{field.Value.ClassName}_{field.Value.FieldName}.g.cs", source);
        });
    }
}
```

### 5.2 自定义分析器

```csharp
[DiagnosticAnalyzer(LanguageNames.CSharp)]
public class AsyncVoidMethodAnalyzer : DiagnosticAnalyzer
{
    private static readonly DiagnosticDescriptor Rule = new(
        "FANDEX001",
        "避免使用 async void 方法",
        "方法 '{0}' 使用了 async void，建议返回 Task",
        "设计",
        DiagnosticSeverity.Warning,
        true);

    public override ImmutableArray<DiagnosticDescriptor> SupportedDiagnostics =>
        [Rule];

    public override void Initialize(AnalysisContext context)
    {
        context.ConfigureGeneratedCodeAnalysis(GeneratedCodeAnalysisFlags.None);
        context.EnableConcurrentExecution();
        context.RegisterSyntaxNodeAction(AnalyzeMethod, SyntaxKind.MethodDeclaration);
    }

    private void AnalyzeMethod(SyntaxNodeAnalysisContext ctx)
    {
        var method = (MethodDeclarationSyntax)ctx.Node;
        if (method.Modifiers.Any(m => m.IsKind(SyntaxKind.AsyncKeyword)) &&
            method.ReturnType is PredefinedTypeSyntax { Keyword.ValueText: "void" })
        {
            ctx.ReportDiagnostic(Diagnostic.Create(
                Rule, method.GetLocation(), method.Identifier.Text));
        }
    }
}
```

## 6. CI/CD

### 6.1 GitHub Actions

```yaml
# .github/workflows/dotnet.yml
name: .NET CI

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Setup .NET
        uses: actions/setup-dotnet@v4
        with:
          dotnet-version: 9.0.x

      - name: Restore
        run: dotnet restore

      - name: Build
        run: dotnet build --no-restore -c Release

      - name: Test
        run: dotnet test --no-build -c Release --logger "trx" --collect:"XPlat Code Coverage"

      - name: Publish Coverage
        uses: codecov/codecov-action@v4
        with:
          files: ./**/coverage.cobertura.xml

      - name: Publish
        run: dotnet publish -c Release -r linux-x64 --self-contained -o ./publish

      - name: Upload Artifact
        uses: actions/upload-artifact@v4
        with:
          name: app
          path: ./publish
```
