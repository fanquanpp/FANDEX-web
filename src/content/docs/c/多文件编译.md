## 1. 头文件

```c
// utils.h
#ifndef UTILS_H
#define UTILS_H

int add(int a, int b);
void print_hello(void);

#endif
```

```c
// utils.c
#include "utils.h"
#include <stdio.h>

int add(int a, int b) { return a + b; }
void print_hello(void) { printf("Hello!\n"); }
```

## 2. 编译

```bash
gcc -c utils.c -o utils.o
gcc -c main.c -o main.o
gcc utils.o main.o -o program
```

## 3. Makefile

```makefile
CC = gcc
CFLAGS = -Wall -Wextra -std=c17

SRCS = main.c utils.c
OBJS = $(SRCS:.c=.o)
TARGET = program

$(TARGET): $(OBJS)
	$(CC) $(CFLAGS) -o $@ $^

%.o: %.c
	$(CC) $(CFLAGS) -c -o $@ $<

clean:
	rm -f $(OBJS) $(TARGET)
```
