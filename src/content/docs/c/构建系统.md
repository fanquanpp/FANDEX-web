## 1. CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.20)
project(MyProject C)

set(CMAKE_C_STANDARD 17)

add_executable(program main.c utils.c)

target_include_directories(program PRIVATE include)
target_link_libraries(program m)
```

## 2. 构建命令

```bash
mkdir build && cd build
cmake ..
cmake --build .
```
