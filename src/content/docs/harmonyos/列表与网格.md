## 1. List

```typescript
List() {
  ForEach(this.data, (item: Data) => {
    ListItem() {
      Row() {
        Text(item.title)
      }
    }
  }, (item: Data) => item.id.toString())
}
.listDirection(Axis.Vertical)
.cachedCount(5
```

## 2. Grid

```typescript
Grid() {
  ForEach(this.items, (item) => {
    GridItem() {
      Text(item.name)
    }
  })
}
.columnsTemplate('1fr 1fr 1fr')
```
